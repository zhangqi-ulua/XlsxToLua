####连接MySQL数据库的参数字符串
connectMySQLString:server=127.0.0.1;port=3306;uid=root;password=root;database=mydb;

####导出Excel文件所在目录
exportExcelPath:ExportExcel

exportDataTableNames:hero|hero_equipment|prop|system

columnBackgroundColor:34,36,40,24

####data表标签按钮背景色的ColorIndex
dataSheetTabColor:43
